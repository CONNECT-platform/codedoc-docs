linkConfiguration

CODEDOC's configuration is read from .codedoc/config.ts. A typical config file looks like this:

.codedoc/config.ts
1import { configuration } from '@codedoc/core';

2

3import { theme } from './theme';

4

5

6export const config = /*#__PURE__*/configuration({

7  theme,

8  dest: {

9    namespace: '/my-project',

10  },

11  page: {

12    title: {

13      base: 'My Project'

14    },

15    favicon: '/favicon.ico'

16  },

17  misc: {

18    github: {

19      user: 'johndoe',

20      repo: 'my-project',

21    },

22    gitter: {

23      room: 'my-project'

24    }

25  }

26});

linkTheme

The theme of your project is configured in a different file, .codedoc/theme.ts. This is because the theme is also imported by client-side codes (codedoc's bundle) and hence should not be mixed up with your typical project configuration, which might include imports and elements that cannot be transported (or should not be transported) to the bundle.

linkAvailable Properties

The following are all of the configurable properties, set to their default values:

.codedoc/config.ts
1import { configuration,

2         DefaultMarkdownCustomComponents,

3         DefaultToCMarkdownCustomComponents,

4       } from '@codedoc/core';

5import { guessTitle } from '@codedoc/core/transport';

6

7import { theme } from './theme';

8

9export const config = configuration({

10  theme,                                 // --> always include your theme

11  src: {                                 // @see /docs/config/entry

12    base: 'docs/md',                     // --> the base folder for all markdowns

13    toc: '_toc.md',                      // --> markdown file for toc, relative to `base`

14    pick: /\.md$/,                       // --> which files to pick (default: .md files)

15    drop: /(^_)|(\/_)/,                  // --> which files to drop (default: _something.md files)

16  },

17

18  dest: {                                // @see /docs/config/output

19    html: '.',                           // --> the base folder for HTML files

20    assets: '.',                         // --> the base folder for assets

21    bundle: 'docs/assets',               // --> where to store codedoc's bundle (relative to `assets`)

22    styles: 'docs/assets',               // --> where to store codedoc's styles (relative to `assets`)

23    namespace: '',                       // --> project namespace

24  },

25

26  bundle: {                              // @see /docs/config/bundle

27    init: [                              // --> a list of initialization scripts for codedoc's bundle

28      codeSelection$,                    // --> this fella makes code snippets interactively selectable

29      sameLineLengthInCodes$,            // --> this fella ensures the same line length in code snippets

30      codeLineHints$,                    // --> this fella is responsible for hints in code snippets

31      codeLineRef$,                      // --> this fella is responsible for references in code snippets

32      smartCopy$,                        // --> this fella makes `CopyButton` work

33      copyHeadings$,                     // --> this fella makes links to headings easily copiable

34      contentNavHighlight$,              // --> this fella highlights current section in content nav

35      deferredIframes$,                  // --> this fella defers iframes for faster loading

36      smoothLoading$,                    // --> this fella makes loading new pages smoothly like an SPA app

37      tocHighlight$,                     // --> this fella highlights the current page in ToC

38      postNavSearch$,                    // --> this fella is repsonsible for searching a term in a page

39    ],

40  },

41

42  dev: {

43    port: 3000                           // --> the port for local dev server

44  },

45

46  page: {                                // @see /docs/config/page

47    title: {                             // --> configuration for page title

48      base: 'My Project',                // --> the base term of page title

49      connector: ' | ',                  // --> the connector of different parts of the page title

50      extractor: (content, config) =>    // --> the page-specific title extractor

51        guessTitle(

52          content, 

53          config.page.title.base, 

54          config.page.title.connector

55        ),

56    },

57    favicon: undefined                   // --> link to your fav icon

58    meta: {                              // --> meta tags of each page

59      subject: undefined                 // --> the subject meta tag for each page

60      description: undefined             // --> description meta tag for each page

61      keywords: [],                      // --> a list of SEO keywords

62      themeColor: undefined,             // --> the browser bar color of your docs

63      appleMobileWebStatusBarStyle:      // --> same as above, but for iOS Safari

64        undefined

65    },

66    fonts: {                             // --> font settings

67      text: {                            // --> font used for texts

68        url:                             // --> URL of font used for texts

69          'https://fonts.googleapis.com/css?family=Hind:400,700&display=swap',

70        name: 'Hind'                     // --> name of font used for texts

71        fallback: 'sans-serif'           // --> the fallback font for texts

72      },

73      code: {                            // --> font used for codes

74        url:                             // --> URL of font used for codes

75          'https://fonts.googleapis.com/css?family=Source+Code+Pro:300,400&display=swap',

76        name: 'Source Code Pro',         // --> name of the font used for codes

77        fallback:                        // --> fallback font for codes

78          `'Courier New', Courier, monospace`

79      },

80      icon: {                            // --> the icon font

81        url:                             // --> url of hte icon font (and perhaps the outline icon font)

82          'https://fonts.googleapis.com/icon?family=Material+Icons|Material+Icons+Outlined',

83        name: 'Material Icons',          // --> name of the icon font

84        outline:                         // --> name of the outline icon font

85          'Material Icons Outlined'

86      }

87    },

88    scripts: [],                         // --> a list of URLs for external scripts to load

89    stylesheets: [],                     // --> a list of URLs for external stylesheets to load

90  },

91

92  markdown: {                            // @see /docs/config/markdown

93    Code,

94    Heading,

95    customComponents:                    // --> custom components available in markdown

96      DefaultMarkdownCustomComponents,   // --> default components provided by codedoc

97  },

98  tocMarkdown: {                         // @see /docs/config/markdown

99    Heading: ToCHeading,

100    customComponents:                    // --> custom components available while parsing toc markdown

101      DefaultToCMarkdownCustomComponents,// --> default toc components provided by codedoc

102  },

103});
